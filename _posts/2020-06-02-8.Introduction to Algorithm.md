  5. Search(x)
  - Linear Search
  - Binary Search

```c++
int binsearch(int l,int r,int key)
{
  int mid = (l+r)/2;
  while(l<r)
  {
    if(A[mid]==key)
    {
      return mid;
    }
    else if(key<A[mid])
      return binsearch(l,mid-1,key);
    else
      return binsearch(mid+1, r,key);
  }
  return -1;
}
```

  6. Get(index)

```c++
int get(int index)
{
  if(index>=0 && index<length)
  {
    return A[index];
  }
}
```

  7. Set(index, x)

```c++
set(index,x)
{
  if(index>=0 && index<length)
  {
    A[index]=x;
  }
}
```

  8. max()/min()

```c++
int max()
{
  max = A[0];
  for(int i = 1; i<length; i++1)
  {
    if(A[i]>max)
      max=A[i];
  }
  return max;
}
```

  9. Reverse()

```c++
for(int i=0, j=Length-1;i<j; i++, j--)
{
  temp=A[i];
  A[i] = A[j]
  A[j] = temp;
}
```

  10. switch()/rotate()

```c++
for (i=0; i<length; i++)
{
  if(key==A[i])
  {
    swap(A[i],A[i-1])
    return i-1;
  }
}
```

  11. sorted

```c++
bool isSorted(int[] A, n)
{
  for(int i=0; i<n-1; i++)
  {
    if(A[i]>A[i+1])
    return false
  }
  return true
}
```
