    double r, a, b, h;

            break;

#include <iostream>

using namespace std;

/* overloading functions */
int power(int, int);
double power(double, int);
int main()
{

    cout << power(2, 6) << endl;

    return 0;
}

/*
    2 ^ 3 = 2 * 2 * 2

    2 ^ 1 = 2
    2 ^ 2 = 4
    2 ^ 3 = 8
*/
int power(int b, int e) // b = 8, e = 1
{
    int tmp = b; //tmp = 2
    int i = 0;

    while(i++ < e)
    {
        cout << tmp << " ^ " << i << " = " << b << endl;
        if (i != e)
            b *= tmp; //b = b * tmp;
    }

    return b;
}
double power(double b, int e)
{
    double tmp = b; //tmp = 2
    while(e-- > 1)
        b *= tmp; //b = b * tmp;

    return b;
}

{
    double tab[5];

    for (int i = 0; i < 5; i++)
    {
        cout << "Input " << i+1 << " number: ";
        cin >> tab[i];
    }

    cout << "The minimum value was: " << minValue(tab) << endl;
    cout << "The maximum value was: " << maxValue(tab) << endl;

    return 0;
}

double minValue(double tab[])
{
    double minValue = tab[0];

    for (int i = 1; i < 5; i++)
    {
        if (minValue > tab[i])
            minValue = tab[i];

    }

    return minValue;
}
double maxValue(double tab[])
{
    double maxValue = tab[0];

    for (int i = 1; i < 5; i++)
    {
        if (maxValue < tab[i])
            maxValue = tab[i];

    }

    return maxValue;
}
